# NameNamespace: par (nombre, namespace) resultante de interpretar el texto que precede
# a ':' o '>>' en la línea de un nodo. El namespace se devuelve siempre en minúsculas.

CLASS NameNamespace
    FIELD name: STRING
    FIELD namespace: STRING

    CONSTRUCTOR (name, namespace)
        name = name
        namespace = namespace
    END CONSTRUCTOR

    FUNCTION getName(): STRING
        RETURN name
    END FUNCTION

    FUNCTION getNamespace(): STRING
        RETURN namespace
    END FUNCTION
END CLASS

# Separa el nombre del nodo de su namespace opcional entre paréntesis.
#
# Entrada (raw_name): el texto anterior a ':' o '>>', p. ej.:
#   "Nombre de nodo"
#   "Nombre de nodo (com.example.docs)"
#   "Nombre de nodo (@un.namespace.especial)"
#
# El formato del namespace (a.b mínimo, ASCII, '@' opcional) NO se valida aquí; se valida
# más adelante al construir el Node (ver validations.txt / node.txt). Aquí sólo se separan
# las dos partes y se normaliza el namespace a minúsculas.
FUNCTION parse(raw_name: STRING, inherited_namespace: STRING, line_number: INTEGER, full_line: STRING): NameNamespace

    IF raw_name IS NULL
        THROW ParseException(line_number, "INVALID_LINE", "Line not valid: " + full_line)
    END IF

    raw_name = raw_name.trim()

    index_open: INTEGER  = raw_name.indexOf("(")
    index_close: INTEGER = raw_name.indexOf(")")

    name: STRING = NULL

    # Namespace por defecto: el heredado del padre, o vacío si no hay
    namespace: STRING = inherited_namespace
    IF namespace IS NULL
        namespace = EMPTY_NAMESPACE
    END IF

    IF index_open != -1 AND index_close != -1
        # Hay paréntesis: el '(' debe ir antes del ')' y ')' debe ser el último carácter
        IF index_open > index_close OR index_close != raw_name.length() - 1
            THROW ParseException(line_number, "INVALID_NAMESPACE", "Line not valid: " + full_line)
        END IF

        name = raw_name.substring(0, index_open).trim()
        namespace = raw_name.substring(index_open + 1, index_close).trim()

        # El namespace vacío "()" no está permitido
        IF namespace.isEmpty()
            THROW ParseException(line_number, "INVALID_NAMESPACE", "Line not valid: " + full_line)
        END IF

    ELSE IF index_open == -1 AND index_close == -1
        # Sin paréntesis: todo el texto es el nombre; se mantiene el namespace heredado
        name = raw_name

    ELSE
        # Sólo un paréntesis (abre sin cerrar o cierra sin abrir): inválido
        THROW ParseException(line_number, "INVALID_NAMESPACE", "Line not valid: " + full_line)
    END IF

    RETURN NameNamespace(name, lowerCase(namespace))

END FUNCTION
